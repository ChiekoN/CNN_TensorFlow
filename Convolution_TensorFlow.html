def create_minibatch(X, Y, batch_size, seed=None):
    """ Create mini batches
        X, Y: input data and labels to be divided into mini batches
        batch_size: number of samples in each mini batch
        seed: random seed for shuffling data
    """
    
    np.random.seed(seed)
    M = X.shape[0]
    batch_num = int(np.floor(M / batch_size))
    
    # Shuffle the sequence of samples
    random_idx = np.random.permutation(M)
    X_random = X[random_idx, :, :, :]
    Y_random = Y[random_idx, :]
    
    minibatches = []
    for i in np.arange(batch_num):
        head = i * batch_size
        X_batch = X_random[head:head+batch_size, :, :, :]
        Y_batch = Y_random[head:head+batch_size, :]
        minibatches.append((X_batch, Y_batch))

    if (M % batch_size) > 0:
        X_batch = X_random[batch_num*batch_size:, :, :, :]
        Y_batch = Y_random[batch_num*batch_size:, :]
        minibatches.append((X_batch, Y_batch))
        
    return minibatches

Define a Convolutinal NN architecture¶

  • Conv2d layer 1 : input shape = (28, 28, 1), filter=(2, 2, 1, 8), strides=1, padding=SAME
  • Activation function 1 : ReLU

  • Max pooling layer 1 : input shape = (28, 28, 8) filter=(4, 4), strides=4, padding=SAME

  • Conv2d layer 2 : input shape = (7, 7, 8), filter=(2, 2, 8, 16), strides=1, padding=SAME

  • Activation function 2 : ReLU
  • Max pooling layer 2 : input shape = (7, 7, 16), filter=(2, 2), stride=2, padding=SAME
  • Fully connected layer : input nodes = (4 *4 *16) = 256, output nodes = (number of classes) = 10

def cnn_model(X, weights, num_classes):
    """ Define CNN architecture.
        X : input data
        weights : weights = [W1, W2] (filters for 2 conv layers)
        num_classes : number of classes
    """
    
    W1 = weights[0]
    W2 = weights[1]
    
    # Build CNN
    Z1 = tf.nn.conv2d(X, W1, strides=[1, 1, 1, 1], padding='SAME')
    A1 = tf.nn.relu(Z1)
    P1 = tf.nn.max_pool(A1, ksize=[1, 4, 4, 1], strides=[1, 4, 4, 1], padding='SAME')
    Z2 = tf.nn.conv2d(P1, W2, strides=[1, 1, 1, 1], padding='SAME')
    A2 = tf.nn.relu(Z2)
    P2 = tf.nn.max_pool(A2, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='SAME')
    P2_flat = tf.contrib.layers.flatten(P2)
    Z3 = tf.contrib.layers.fully_connected(P2_flat, num_classes, activation_fn=None)
    
    # return without applying softmax function
    return Z3

def model_train(X_train, Y_train, X_test, Y_test, filename, epochs=100, batch_size=64, learning_rate=0.01):
    """ Run the graph, train and test the model.
        X_train, Y_train : training data
        X_test, Y_test : testing data
        filename : file name to save Tensorflow variables
        epochs : number of epochs
        batch_size : number of samples in each batch
        learning_rate : learning rate for Adam optimizer
    """
    
    tf.reset_default_graph()
    

    # create placeholders for X, Y
    _, height, width, _ = X_train.shape
    classes = Y_train.shape[1]
    print("Height, width, classes = {}, {}, {}".format(height, width, classes))
    
    # define input and output placeholders
    X = tf.placeholder(tf.float32, shape=[None, height, width, 1])
    Y = tf.placeholder(tf.float32, shape=[None, classes])

    # initialize weights = filters[height, width, channels, number of filters]
    W1 = tf.get_variable("W1", [2, 2, 1, 8], initializer=tf.contrib.layers.xavier_initializer(seed=10))
    W2 = tf.get_variable("W2", [2, 2, 8, 16], initializer=tf.contrib.layers.xavier_initializer(seed=10))
    weights = [W1, W2]
    
    # feed forward
    Z3 = cnn_model(X, weights, classes)

    # loss = Tensor(minibatch_size, 1)
    loss = tf.nn.softmax_cross_entropy_with_logits_v2(labels=Y, logits=Z3)
    # cost = one number, the average of the mini batch
    cost = tf.reduce_mean(loss)

    # optimizer (and backpropagation)
    optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate).minimize(cost)
    
    init = tf.global_variables_initializer()

    # Saver to save the model after training
    saver = tf.train.Saver()
    
    # Start the session
    with tf.Session() as sess:
        sess.run(init)

        costs = []
        for epoch in range(epochs):

            minibatches = create_minibatch(X_train, Y_train, batch_size)

            all_batch_cost = 0.
            batch_num = len(minibatches)

            for batch in minibatches:
                (X_batch, Y_batch) = batch
                batch_cost, _ = sess.run([cost, optimizer], feed_dict={X:X_batch, Y:Y_batch})
                all_batch_cost += batch_cost

            # calculate avarage cost for minibatches
            cost_for_epoch = all_batch_cost / batch_num
            print("Epoch {} : cost = {}".format(epoch, cost_for_epoch))
            costs.append(cost_for_epoch)
    
        # Calculate accuracy
        predict_op = tf.argmax(Z3, axis=1)
        prediction = tf.equal(predict_op, tf.argmax(Y, axis=1))
        accuracy = tf.reduce_mean(tf.cast(prediction, tf.float32))
        # Input training data into the model and calculate accuracy from the prediction
        train_accuracy = accuracy.eval({X:X_train, Y:Y_train})
        # Input testing data into the model and calculate accuracy
        test_accuracy = accuracy.eval({X:X_test, Y:Y_test})
        
        saver.save(sess, filename)
        
    return train_accuracy, test_accuracy, costs

def model_predict(X_new, num_classes, filename):
    """ Predict classes of new data.
        X_new : input data to predict
        num_classes : number of classes
        filename : file name from which saved variables(weights) are loaded
        """
    
    tf.reset_default_graph()
    
    
    (_, height, width, _) = X_new.shape

    X = tf.placeholder(tf.float32, shape=[None, height, width, 1])

    W1 = tf.get_variable("W1", [2, 2, 1, 8], initializer=tf.contrib.layers.xavier_initializer(seed=10))
    W2 = tf.get_variable("W2", [2, 2, 8, 16], initializer=tf.contrib.layers.xavier_initializer(seed=10))

    Z3 = cnn_model(X, [W1, W2], num_classes)
    pred = tf.argmax(Z3, axis=1)
    
    saver =  tf.train.Saver()
    
    
    with tf.Session() as sess:
        # Restore variables from saver
        saver.restore(sess, filename)
        # Get variable names in this graph
        var_list = tf.global_variables()
        var_names = [var_list[i].name for i in range(len(var_list))]
        print("Restored variables = {}".format(var_names))
        
        sess.run(var_names)
        
        # Predict the class
        pred_result = pred.eval({X:X_new})
                
            
    return pred_result
